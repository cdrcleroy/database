CREATE TABLE categorie(
   id_cat BYTE,
   categorie VARCHAR(50) NOT NULL,
   PRIMARY KEY(id_cat)
);

CREATE TABLE produit(
   id_prod BYTE,
   nom_prod VARCHAR(50),
   prix_kg CURRENCY,
   id_cat BYTE NOT NULL,
   PRIMARY KEY(id_prod),
   FOREIGN KEY(id_cat) REFERENCES categorie(id_cat)
);

CREATE TABLE client(
   id_client INT,
   nom_client VARCHAR(50),
   prenom VARCHAR(50),
   PRIMARY KEY(id_client)
);

CREATE TABLE date_vente(
   quantieme DATE,
   PRIMARY KEY(quantieme)
);

CREATE TABLE vendre(
   id_prod BYTE,
   id_client INT,
   quantieme DATE,
   quantité INT,
   PRIMARY KEY(id_prod, id_client, quantieme),
   FOREIGN KEY(id_prod) REFERENCES produit(id_prod),
   FOREIGN KEY(id_client) REFERENCES client(id_client),
   FOREIGN KEY(quantieme) REFERENCES date_vente(quantieme)
);


categorie = (id_cat BYTE, categorie VARCHAR(50));
produit = (id_prod BYTE, nom_prod VARCHAR(50), prix_kg CURRENCY, #id_cat);
client = (id_client INT, nom_client VARCHAR(50), prenom VARCHAR(50));
date_vente = (quantieme DATE);
vendre = (#id_prod, #id_client, #quantieme, quantité INT);
